function solution(n, k, cmd) {
  const table = Array.from({ length: n }, (_, i) => i);
  const answerArr = Array.from({ length: n }, (_) => "X");
  let cursor = k;
  const stack = [];

  for (const c of cmd) {
    if (c.startsWith("U")) {
      const amount = c.at(-1);
      cursor = clamp(0, cursor - Number(amount), table.length);
    } else if (c.startsWith("D")) {
      const amount = c.at(-1);
      cursor = clamp(0, cursor + Number(amount), table.length);
    } else if (c.startsWith("C")) {
      stack.push(table[cursor]);
      table.splice(cursor, 1);
      if (cursor === table.length - 1) {
        cursor -= 1;
      }
    } else {
      const top = stack.pop();
      table.push(top);
      table.sort((a, b) => a - b);
    }
  }

  for (const t of table) {
    answerArr[t] = "O";
  }
  return answerArr.join("");
}

  const table = Array.from({ length: n }, (_, i) => i);
  const answerArr = Array.from({ length: n }, (_) => "X");
  let cursor = k;
  const stack = [];

  for (const c of cmd) {
    if (c.startsWith("U")) {
      const amount = c.split(" ")[1];
      cursor = cursor - Number(amount);
    } else if (c.startsWith("D")) {
      const amount = c.split(" ")[1];
      cursor = cursor + Number(amount);
    } else if (c.startsWith("C")) {
      stack.push(table[cursor]);
      table.splice(cursor, 1);
      if (cursor === table.length) {
        cursor -= 1;
      }

      const top = stack.pop();
      table.push(top);
      table.sort((a, b) => a - b);
      const findIndex = table.findIndex((t) => t === top);
      if (findIndex <= cursor) cursor += 1;
    }
  }

  for (const t of table) {
    answerArr[t] = "O";
  }
  return answerArr.join("");
}
